pragma solidity ^0.8.2;
 
import "@openzeppelin/contracts-upgradeable/token/ERC1155/ERC1155Upgradeable.sol";
import "@openzeppelin/contracts-upgradeable/access/OwnableUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/security/PausableUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/token/ERC1155/extensions/ERC1155SupplyUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/utils/CountersUpgradeable.sol";
import "@openzeppelin/contracts-upgradeable/proxy/utils/Initializable.sol";
import "@openzeppelin/contracts-upgradeable/utils/StringsUpgradeable.sol";
import "./IHolyDogsUpgradeable.sol";
 
contract HolyDogsUpgradeable is Initializable, ERC1155Upgradeable, OwnableUpgradeable, PausableUpgradeable, ERC1155SupplyUpgradeable {
    
    string public name;
 
    string public symbol;
 
    uint256 public initTime;
 
    uint256 public mintPrice;
 
    bool private isTimeout;
 
    uint8 constant NFT_AMOUNT = 1;
 
    uint16 constant MAX_SUPPLY = 999;
 
    uint16 public totalSupply;
 
    string baseUri;
 
    using CountersUpgradeable for CountersUpgradeable.Counter;
 
    CountersUpgradeable.Counter private _tokenIdCounter;
 
    function initialize(
        string memory nameOfContract, 
        string memory symbolOfContract
    ) initializer external {
        __ERC1155_init("");
        __Ownable_init();
        __Pausable_init();
        __ERC1155Supply_init();
        name = nameOfContract;
        symbol = symbolOfContract;
        initTime = block.timestamp;
    }
 
    function setURI(string memory newuri) public onlyOwner {
        baseUri = newuri;
    }
 
    function uri(uint id) public view virtual override(ERC1155Upgradeable) whenNotPaused returns (string memory) 
    {
        return string(abi.encodePacked(baseUri,StringsUpgradeable.toString(id)));
    }
 
    function setTokenPrice (uint256 mintValue) public onlyOwner whenNotPaused
    {
        mintPrice = mintValue;
    }
 
    function mint() external payable
    {
        require(mintPrice > 0);
        require(MAX_SUPPLY > totalSupply, "Maximum supply is exceed");
        if (block.timestamp >= initTime + 60 days) {
            _airDrop();
        }
        require(isTimeout == false, "HolyDogs Upgradeable: Contract has been deactivated");
        require(msg.value >= mintPrice, "HolyDogs Upgradeable: Insufficient ETH supplied");
        uint256 tokenId = _tokenIdCounter.current();        
        _tokenIdCounter.increment();
        _mint(msg.sender, tokenId, NFT_AMOUNT, "");
    }
 
    function _airDrop() internal {
        uint16 remainingToken  = MAX_SUPPLY - totalSupply;
        for (uint16 i = 0; i < remainingToken; i++ ) {
            uint256 tokenId = _tokenIdCounter.current();            
            _tokenIdCounter.increment();
            _mint(owner(), tokenId, NFT_AMOUNT, "");
            isTimeout = true;
        }
    }
 
    function pause() external onlyOwner 
    {
        _pause();
    }
 
    function unpause() external onlyOwner 
    {
        _unpause();
    }
 
    function _beforeTokenTransfer(address operator, address from, address to, uint256[] memory ids, uint256[] memory amounts, bytes memory data)
        internal
        whenNotPaused
        override(ERC1155Upgradeable, ERC1155SupplyUpgradeable)
    {
        super._beforeTokenTransfer(operator, from, to, ids, amounts, data);
    }
}
