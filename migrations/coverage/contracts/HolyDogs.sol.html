// SPDX-License-Identifier: MIT
pragma solidity ^0.8.2;
 
import "@openzeppelin/contracts/token/ERC1155/ERC1155.sol";
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/Pausable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";
import "@openzeppelin/contracts/utils/Strings.sol";
import "./Random.sol";
 
contract HolyDogs is ERC1155, Ownable, Pausable, ReentrancyGuard, Random {
 
    string public name;
 
    string public symbol;
 
    uint256 public mintEndTime;
 
    uint16 public totalSupply;
 
    uint256 public mintPrice;
 
    uint8 constant NFT_AMOUNT = 1;
 
    uint16 constant MAX_SUPPLY = 999;
 
    string public baseUri;
 
      // Zero Address
    address constant ZERO_ADDRESS = address(0);
 
    constructor(
        string memory _name, 
        string memory _symbol,
        string memory _uri,
        uint256 _mintEndTime,
        uint256 _mintPrice
        
    ) ERC1155("") Random(MAX_SUPPLY) {
        name = _name;
        symbol = _symbol;
        mintEndTime = _mintEndTime;
        baseUri = _uri;
        mintPrice = _mintPrice;
    }
 
    function updateBaseURI(string memory newBaseUri) external onlyOwner whenNotPaused {
        require(keccak256(abi.encodePacked(baseUri)) != keccak256(abi.encodePacked(newBaseUri)) , "HolyDogs: Current state is already what you have selected.");
        baseUri = newBaseUri;
    }
 
    function updateEndTime(uint256 newMintEndTime) external onlyOwner whenNotPaused {
        require(keccak256(abi.encodePacked(mintEndTime)) != keccak256(abi.encodePacked(newMintEndTime)) , "HolyDogs: Current state is already what you have selected.");
        require(block.timestamp <= mintEndTime , "HolyDogs: Mint process is ended");
        mintEndTime = newMintEndTime;
    }
 
    function uri(uint id) public view virtual override(ERC1155) whenNotPaused returns (string memory) 
    {
        return string(abi.encodePacked(baseUri,Strings.toString(id)));
    }
 
    function updateMintPrice (uint256 newMintPrice) public onlyOwner whenNotPaused
    {
        require(keccak256(abi.encodePacked(mintPrice)) != keccak256(abi.encodePacked(newMintPrice)) , "HolyDogs: Current state is already what you have selected.");
        mintPrice = newMintPrice;
    }
 
    function mint() external payable whenNotPaused nonReentrant
    {
        require(MAX_SUPPLY > totalSupply, "HolyDogs: Maximum supply is exceed");
        require(block.timestamp <= mintEndTime , "HolyDogs: Mint process is ended");
        require(msg.value >= mintPrice, "HolyDogs: Insufficient ETH supplied");
        uint256 tokenId = _getRandomNFTTokenID();
        totalSupply++;
        _mint(msg.sender, tokenId,NFT_AMOUNT, "");
    }
 
    function airDrop(address account) external onlyOwner whenNotPaused {
        require(MAX_SUPPLY > totalSupply, "HolyDogs: Maximum supply is exceed");
        uint256 tokenId = _getRandomNFTTokenID();
        _mint(account, tokenId, NFT_AMOUNT, "");
 
    }
 
    function airDrop(address account, uint256 amount) external onlyOwner whenNotPaused {
        for(uint16 i=0; i<amount; i++){
            require(MAX_SUPPLY > totalSupply, "HolyDogs: Maximum supply is exceed");
            uint256 tokenId = _getRandomNFTTokenID();
            totalSupply++;
            _mint(account, tokenId, NFT_AMOUNT, "");
        }
    }
 
    function withdraw(address payable account , uint256 amount) external onlyOwner whenNotPaused nonReentrant {
        require(account != ZERO_ADDRESS, "HolyDogs: Cannot withdraw to Zero Address");
        payable(account).transfer(amount);
    }
 
    function pause() public onlyOwner {
        _pause();
    }
 
    function unpause() public onlyOwner {
        _unpause();
    }
 
    function _beforeTokenTransfer(address operator, address from, address to, uint256[] memory ids, uint256[] memory amounts, bytes memory data)
        internal
        whenNotPaused
        override(ERC1155)
    {
        super._beforeTokenTransfer(operator, from, to, ids, amounts, data);
    }
}
